# Use official Tomcat image
FROM tomcat:9.0-jdk17

# Remove default webapps to avoid the ROOT app
RUN rm -rf /usr/local/tomcat/webapps/*

# Copy your WAR into Tomcat webapps as ROOT
COPY lms.war /usr/local/tomcat/webapps/ROOT.war

# Expose Tomcat port
EXPOSE 8080

# Start Tomcat
CMD ["catalina.sh", "run"]
